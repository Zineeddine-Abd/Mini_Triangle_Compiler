! This is a test program in Mini-Triangle language
! It calculates the factorial of a number

let
  var n : int;
  var fact : int;
  var i : int
in
  begin
    n := 5;
    fact := 1;
    i := 1;

    while i <= n do
      begin
        fact := fact * i;
        i := i + 1
      end;

    ! Now fact contains 5! (120)

    if fact = 120 then
      fact := fact
    else
      fact := 0
  end
